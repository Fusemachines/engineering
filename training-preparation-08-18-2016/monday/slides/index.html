AJAX

mozilla: It is the use of the XMLHttpRequest object to communicate with server-side scripts. 

 // request data from a server
makeRequest(url) {
  // promise provides a function (then) that execute user's code 
  // whenever either resolve or reject is called
  return new Promise((resolve, reject) => {
    let httpRequest = new XMLHttpRequest();

    // readystatechange is an event
    // onreadystatechange is the callback function
    httpRequest.onreadystatechange = ({currentTarget, target}) => {

      resolve(JSON.parse(httpRequest.responseText || '{}'));

    };
    httpRequest.open('GET', url);
    httpRequest.send();

  });
} 
// NOTE that XMLHttpRequest does not work on some browsers.
// so it is better to use jQuery.ajax instead!
